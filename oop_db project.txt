    private void insertActionPerformed(java.awt.event.ActionEvent evt) {                                       
        try{
            Class.forName(Driver);
            Connection con =DriverManager.getConnection(url, user, pass);
            String sql = "insert into Professors "
                    +"(P_SSN,P_name,P_rank,P_Age,Research_specialty)"
                    +"values(?,?,?,?,?)";
            PreparedStatement stm = con.prepareStatement(sql);
            stm.setString(1,P_SSN.getText());
            stm.setString(2,P_name.getText());
            stm.setString(3,P_rank.getText());
            stm.setString(4,P_Age.getText());
            stm.setString(5,Research_specialty.getText());
            
            stm.executeUpdate();
            JOptionPane.showMessageDialog(this, "inserted");
            
        }catch(Exception e){
            JOptionPane.showMessageDialog(this, e.getMessage());
        }
    }

    private void SearchActionPerformed(java.awt.event.ActionEvent evt) {                                       
       try{
            Class.forName(Driver);
            Connection con =DriverManager.getConnection(url, user, pass);  
            String sql = "select * from Professors where P_SSN="+Integer.parseInt(P_SSN.getText());          
            Statement st = con.createStatement();
            ResultSet rs = st.executeQuery(sql);  
            while(rs.next()){
                String P_SSN = String.valueOf(rs.getInt("P_SSN"));
                String P_name = rs.getString("P_name");
                String P_rank = rs.getString("P_rank");
                String P_Age = String.valueOf(rs.getInt("P_Age"));
                String Research_specialty = rs.getString("Research_specialty"); 
                String tbdata [] = {P_SSN,P_name,P_rank,P_Age,Research_specialty};
                DefaultTableModel tb1Model = (DefaultTableModel) MyTable.getModel();
                tb1Model.addRow(tbdata);
            }           
        }catch(Exception e){
            JOptionPane.showMessageDialog(this, e.getMessage());
        }
    }
